74 std::vector<double>
75 gpmp::ml::AutoEncoder::forward(const std::vector<double> &input) {
76  // forward passes
77  std::vector<double> hidden(hidden_size);
78  std::vector<double> output(output_size);
79 
80  // calculate hidden layer values
81  for (int i = 0; i < hidden_size; ++i) {
82  hidden[i] = 0;
83  for (int j = 0; j < input_size; ++j) {
84  hidden[i] += input[j] * weights_input_hidden[j][i];
85  }
86  hidden[i] = sigmoid({hidden[i]})[0];
87  }
88 
89  // calculate output layer values
90  for (int i = 0; i < output_size; ++i) {
91  output[i] = 0;
92  for (int j = 0; j < hidden_size; ++j) {
93  output[i] += hidden[j] * weights_hidden_output[j][i];
94  }
95  output[i] = sigmoid({output[i]})[0];
96  }
97 
98  return output;
99 }

101 void gpmp::ml::AutoEncoder::train(
102  const std::vector<std::vector<double>> &training_data,
103  int epochs) {
104  for (int epoch = 0; epoch < epochs; ++epoch) {
105  for (const auto &input : training_data) {
106  // forward pass
107  std::vector<double> hidden(hidden_size);
108  std::vector<double> output(output_size);
109 
110  // calculate hidden layer values
111  for (int i = 0; i < hidden_size; ++i) {
112  hidden[i] = 0;
113  for (int j = 0; j < input_size; ++j) {
114  hidden[i] += input[j] * weights_input_hidden[j][i];
115  }
116  hidden[i] = sigmoid({hidden[i]})[0];
117  }
118 
119  // calculate output layer values
120  for (int i = 0; i < output_size; ++i) {
121  output[i] = 0;
122  for (int j = 0; j < hidden_size; ++j) {
123  output[i] += hidden[j] * weights_hidden_output[j][i];
124  }
125  output[i] = sigmoid({output[i]})[0];
126  }
127 
128  // backward pass (gradient descent)
129  for (int i = 0; i < output_size; ++i) {
130  for (int j = 0; j < hidden_size; ++j) {
131  weights_hidden_output[j][i] -=
132  learning_rate * (output[i] - input[i]) * hidden[j];
133  }
134  }
135 
136  for (int i = 0; i < hidden_size; ++i) {
137  for (int j = 0; j < input_size; ++j) {
138  double error = 0;
139  for (int k = 0; k < output_size; ++k) {
140  error += (output[k] - input[k]) *
141  weights_hidden_output[i][k];
142  }
143  weights_input_hidden[j][i] -= learning_rate * error *
144  input[j] * (1 - hidden[i]) *
145  hidden[i];
146  }
147  }
148  }
149  }
150 }
