128 gpmp::ml::AutoEncoder::forward(const std::vector<double> &input) {
129  // forward passes
130  std::vector<double> hidden(hidden_size);
131  std::vector<double> output(output_size);
132 
133  // calculate hidden layer values
134  for (int i = 0; i < hidden_size; ++i) {
135  hidden[i] = 0;
136  for (int j = 0; j < input_size; ++j) {
137  hidden[i] += input[j] * weights_input_hidden[j][i];
138  }
139  hidden[i] = sigmoid({hidden[i]})[0];
140  }
141 
142  // calculate output layer values
143  for (int i = 0; i < output_size; ++i) {
144  output[i] = 0;
145  for (int j = 0; j < hidden_size; ++j) {
146  output[i] += hidden[j] * weights_hidden_output[j][i];
147  }
148  output[i] = sigmoid({output[i]})[0];
149  }
150 
151  return output;

154 void gpmp::ml::AutoEncoder::train(
155  const std::vector<std::vector<double>> &training_data,
156  int epochs) {
157  for (int epoch = 0; epoch < epochs; ++epoch) {
158  for (const auto &input : training_data) {
159  // forward pass
160  std::vector<double> hidden(hidden_size);
161  std::vector<double> output(output_size);
162 
163  // calculate hidden layer values
164  for (int i = 0; i < hidden_size; ++i) {
165  hidden[i] = 0;
166  for (int j = 0; j < input_size; ++j) {
167  hidden[i] += input[j] * weights_input_hidden[j][i];
168  }
169  hidden[i] = sigmoid({hidden[i]})[0];
170  }
171 
172  // calculate output layer values
173  for (int i = 0; i < output_size; ++i) {
174  output[i] = 0;
175  for (int j = 0; j < hidden_size; ++j) {
176  output[i] += hidden[j] * weights_hidden_output[j][i];
177  }
178  output[i] = sigmoid({output[i]})[0];
179  }
180 
181  // backward pass (gradient descent)
182  for (int i = 0; i < output_size; ++i) {
183  for (int j = 0; j < hidden_size; ++j) {
184  weights_hidden_output[j][i] -=
185  learning_rate * (output[i] - input[i]) * hidden[j];
186  }
187  }
188 
189  for (int i = 0; i < hidden_size; ++i) {
190  for (int j = 0; j < input_size; ++j) {
191  double error = 0;
192  for (int k = 0; k < output_size; ++k) {
193  error += (output[k] - input[k]) *
194  weights_hidden_output[i][k];
195  }
196  weights_input_hidden[j][i] -= learning_rate * error *
197  input[j] * (1 - hidden[i]) *
198  hidden[i];
199  }
200  }
201  }
202  }

233 void gpmp::ml::SparseAutoEncoder::train(
234  const std::vector<std::vector<double>> &training_data,
235  int epochs) {
236 
237  const double SPARSITY_TARGET_DECAY = 0.1;
238 
239  for (int epoch = 0; epoch < epochs; ++epoch) {
240  for (const auto &current_input : training_data) {
241  // forward pass
242  std::vector<double> hidden = forward(current_input);
243 
244  // backward pass (gradient descent)
245  for (int i = 0; i < output_size; ++i) {
246  for (int j = 0; j < hidden_size; ++j) {
247  weights_hidden_output[j][i] -=
248  learning_rate * (hidden[i] - current_input[i]) *
249  hidden[j];
250  }
251  }
252 
253  for (int i = 0; i < hidden_size; ++i) {
254  for (int j = 0; j < input_size; ++j) {
255  double error = 0;
256  for (int k = 0; k < output_size; ++k) {
257  error += (hidden[k] - current_input[k]) *
258  weights_hidden_output[i][k];
259  }
260 
261  double sparsity_term =
262  sparsity_weight * (sparsity_target - hidden[i]);
263 
264  weights_input_hidden[j][i] -=
265  learning_rate * (error + sparsity_term) *
266  current_input[j] * (1 - hidden[i]) * hidden[i];
267  }
268  }
269 
270  for (int i = 0; i < hidden_size; ++i) {
271  double average_activation = 0.0;
272  for (const auto &input : training_data) {
273  std::vector<double> current_hidden = forward(input);
274  average_activation += current_hidden[i];
275  }
276  average_activation /= training_data.size();
277  sparsity_target =
278  (1.0 - SPARSITY_TARGET_DECAY) * sparsity_target +
279  SPARSITY_TARGET_DECAY * average_activation;
280  }
281  }
282  }
283 }
