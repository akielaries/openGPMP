46 std::vector<std::unordered_map<std::string, std::string>>
47  csv_read(std::string filename,
48  std::vector<std::string> selected_columns = {}) {
49  std::ifstream file(filename);
50  std::vector<std::unordered_map<std::string, std::string>> data;
51 
52  if (file) {
53  // Read header line
54  std::string line;
55  std::getline(file, line);
56  std::istringstream header(line);
57  std::vector<std::string> column_names;
58  size_t index = 0;
59  std::string name;
60  while (std::getline(header, name, ',')) {
61  // If no columns are selected or the column is selected, add it
62  // to the column names
63  if (selected_columns.empty() ||
64  std::find(selected_columns.begin(), selected_columns.end(),
65  name) != selected_columns.end()) {
66  column_names.push_back(name);
67  index++;
68  }
69  }
70 
71  // Create column_indices map based on column_names
72  std::unordered_map<std::string, size_t> column_indices;
73  for (size_t i = 0; i < column_names.size(); i++) {
74  column_indices[column_names[i]] = i;
75  }
76 
77  // Read data lines
78  while (std::getline(file, line)) {
79  std::istringstream record(line);
80  std::unordered_map<std::string, std::string> row;
81  for (const auto &name : column_names) {
82  std::string value;
83  std::getline(record, value, ',');
84  row[name] = value;
85  }
86  data.push_back(row);
87  }
88  }
89 
90  return data;
91 }

101  // Get the list of column names
102  std::vector<std::string> column_names;
103  std::unordered_map<std::string, size_t> column_widths;
104  for (const auto &row : data) {
105  for (const auto &pair : row) {
106  if (std::find(column_names.begin(), column_names.end(),
107  pair.first) == column_names.end()) {
108  column_names.push_back(pair.first);
109  column_widths[pair.first] = pair.first.size() + 2;
110  }
111  if (pair.second.size() + 2 > column_widths[pair.first]) {
112  column_widths[pair.first] = pair.second.size() + 2;
113  }
114  }
115  }

117  // Print the data
