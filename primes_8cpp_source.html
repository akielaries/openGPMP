147  if (n % 4 == 3)
148  ans = -ans; // (-1/n) = -1 if n = 3 (mod 4)

165  }

257 int64_t mtpk::Primality::pollard_rho(int64_t n) {
258  /* initialize random seed */
259  srand(time(NULL));
260 
261  /* no prime divisor for 1 */
262  if (n == 1)
263  return n;
264 
265  /* even number means one of the divisors is 2 */
266  if (n % 2 == 0)
267  return 2;
268 
269  /* we will pick from the range [2, N) */
270  int64_t x = (rand() % (n - 2)) + 2;
271  int64_t y = x;
272 
273  /* the constant in f(x).
274  * Algorithm can be re-run with a different c
275  * if it throws failure for a composite. */
276  int64_t c = (rand() % (n - 1)) + 1;
277 
278  /* Initialize candidate divisor (or result) */
279  int64_t divisor = 1;
280 
281  /* until the prime factor isn't obtained.
282  If n is prime, return n */
283  while (divisor == 1) {
284  /* Tortoise Move: x(i+1) = f(x(i)) */
285  // take power
286  // calculate modulus
287  x = (mod_pow(x, 2, n) + c + n) % n;
288 
289  /* Hare Move: y(i+1) = f(f(y(i))) */
290  y = (mod_pow(y, 2, n) + c + n) % n;
291  y = (mod_pow(y, 2, n) + c + n) % n;
292 
293  /* check gcd of |x-y| and n */
294  divisor = ba.op_gcd(abs(x - y), n);
295  //divisor = std::__gcd(abs(x - y), n);
296  /* retry if the algorithm fails to find prime factor
297  * with chosen x and c */
298  if (divisor == n)
299  return pollard_rho(n);
300  }
301 
302  return divisor;
303 }
