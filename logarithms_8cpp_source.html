59 uint64_t mtpk::Logarithms::pollard_rho_log(uint64_t g, uint64_t y,
60  uint64_t p) {
61  uint64_t x = rand() % (p - 1) + 1;
62 
63 #ifdef USE_SRAND
64  srand(time(NULL));
65  x = rand() % (p - 1) + 1;
66 #endif
67 
68  uint64_t y1 = y;
69  uint64_t y2 = y;
70  uint64_t d = 1;
71  std::unordered_map<uint64_t, uint64_t> values;
72 
73  // Compute the iterated values of the sequence x_n = g^x_n-1 mod p
74  // until a collision is found
75  while (d == 1) {
76  x = (x * x + 1) % p;
77  y1 = (y1 * g) % p;
78  y2 = (y2 * g) % p;
79  y2 = (y2 * g) % p;
80  d = std::gcd(abs(y2 - y1), p);
81  values[x] = y1;
82  }
83 
84  // Compute the discrete logarithm using the collision point
85  if (d == p) {
86  return -1;
87  } else {
88  uint64_t x0 = abs(y2 - y1) / d;
89  uint64_t y0 = values[x0];
90  uint64_t k = 1;
91  while (y0 != y) {
92  y0 = (y0 * g) % p;
93  k++;
94  }
95  return k * x0 % (p - 1);
96  }
97 }

100 uint64_t mtpk::Logarithms::BSGS(uint64_t a, uint64_t b, uint64_t m) {
101  uint64_t n = (uint64_t)sqrt(m) + 1;
102 
103  std::unordered_map<uint64_t, uint64_t> value;
104 
105  // store all values of a^(n*i) of LHS
106  for (uint64_t i = n; i >= 1; --i) {
107  value[__LOG_PRIMES__.mod_pow(a, i * n, m)] = i;
108  }
109 
110  for (uint64_t j = 0; j < n; ++j) {
111  // calculate (a ^ j) * b and check
112  uint64_t cur = (__LOG_PRIMES__.mod_pow(a, j, m) * b) % m;
113 
114  // If collision occurs i.e., LHS = RHS
115  if (value[cur]) {
116  uint64_t ans = value[cur] * n - j;
117  // Check whether ans lies below m or not
118  if (ans < m) {
119  return ans;
120  }
121  }
122  }
123  return -1;
124 }
