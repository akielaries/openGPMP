59uint64_t mtpk::Logarithms::pollard_rho_log(uint64_t g, uint64_t y, uint64_t p) {
60 uint64_t x = rand() % (p - 1) + 1;
61
62#ifdef USE_SRAND
63 srand(time(NULL));
64 x = rand() % (p - 1) + 1;
65#endif
66
67 uint64_t y1 = y;
68 uint64_t y2 = y;
69 uint64_t d = 1;
70 std::unordered_map<uint64_t, uint64_t> values;
71
72 // Compute the iterated values of the sequence x_n = g^x_n-1 mod p
73 // until a collision is found
74 while (d == 1) {
75 x = (x * x + 1) % p;
76 y1 = (y1 * g) % p;
77 y2 = (y2 * g) % p;
78 y2 = (y2 * g) % p;
79 d = std::gcd(abs(y2 - y1), p);
80 values[x] = y1;
81 }
82
83 // Compute the discrete logarithm using the collision point
84 if (d == p) {
85 return -1;
86 } else {
87 uint64_t x0 = abs(y2 - y1) / d;
88 uint64_t y0 = values[x0];
89 uint64_t k = 1;
90 while (y0 != y) {
91 y0 = (y0 * g) % p;
92 k++;
93 }
94 return k * x0 % (p - 1);
95 }
96}

99uint64_t mtpk::Logarithms::BSGS(uint64_t a, uint64_t b, uint64_t m) {
100 uint64_t n = (uint64_t)sqrt(m) + 1;
101
102 std::unordered_map<uint64_t, uint64_t> value;
103
104 // store all values of a^(n*i) of LHS
105 for (uint64_t i = n; i >= 1; --i) {
106 value[__LOG_PRIMES__.mod_pow(a, i * n, m)] = i;
107 }
108
109 for (uint64_t j = 0; j < n; ++j) {
110 // calculate (a ^ j) * b and check
111 uint64_t cur = (__LOG_PRIMES__.mod_pow(a, j, m) * b) % m;
112
113 // If collision occurs i.e., LHS = RHS
114 if (value[cur]) {
115 uint64_t ans = value[cur] * n - j;
116 // Check whether ans lies below m or not
117 if (ans < m) {
118 return ans;
119 }
120 }
121 }
122 return -1;
123}
