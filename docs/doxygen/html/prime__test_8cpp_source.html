56// Calculates (a * b) mod m
57uint64_t mtpk::PrimalityTest::mod_mul(uint64_t a, uint64_t b, uint64_t m) {
58 uint64_t res = 0;
59 while (b > 0) {
60 if (b & 1) {
61 res = (res + a) % m;
62 }
63 a = (2 * a) % m;
64 b >>= 1;
65 }
66 return res;
67}

70uint64_t mtpk::PrimalityTest::mod_pow(uint64_t a, uint64_t b, uint64_t m) {
71 uint64_t res = 1;
72 a %= m;
73 while (b > 0) {
74 if (b & 1) {
75 res = mod_mul(res, a, m);
76 }
77 a = mod_mul(a, a, m);
78 b >>= 1;
79 }
80 return res;

114/*
115 * determining if a given number is likely to be prime
116 */
117bool mtpk::PrimalityTest::compute_miller_rabin(uint64_t d, uint64_t n) {
118 // Pick a random number in [2..n-2] Corner cases make sure that n
119 // > 4
120 uint64_t a = 2 + rand() % (n - 4);
121 /*
122 std::random_device rd;
123 std::mt19937_64 gen(rd());
124 std::uniform_int_distribution<uint64_t> dis(2, n - 4);
125 uint64_t a = dis(gen);
126 */
127
128 // Compute a^d % n
129 uint64_t x = mod_pow(a, d, n);
130
131 if (x == 1 || x == n - 1) {
132 return true;
133 }
134
135 // Keep squaring x while one of the following doesn't
136 // happen
137 // (I) d does not reach n-1
138 // (II) (x^2) % n is not 1
139 // (III) (x^2) % n is not n-1
140 while (d != n - 1) {
141 x = (x * x) % n;
142 d *= 2;
143
144 if (x == 1) {
145 return false;
146 }
147 if (x == n - 1) {
148 return true;
149 }
150 }
151
152 // Return composite
153 return false;
154}

156bool mtpk::PrimalityTest::witness(uint64_t n, uint64_t d, uint64_t a,
157 uint64_t s) {
158 uint64_t x = mod_pow(a, d, n);
159 if (x == 1 || x == n - 1) {
160 return false;
161 }
162 for (uint64_t r = 1; r < s; r++) {
163 x = mod_mul(x, x, n);
164 if (x == n - 1) {
165 return false;
166 }
167 }
168 return true;
169}

214bool mtpk::PrimalityTest::AKS(uint64_t n) {
215 // Check if n is a perfect power
216 for (uint64_t b = 2; b <= std::sqrt(n); b++) {
217 uint64_t a = std::round(std::pow(n, 1.0 / b));
218 if (std::pow(a, b) == n) {
219 return false;
220 }
221 }
222
223 // Find the smallest r such that ord_r(n) > log2(n)^2
224 uint64_t r = 2;
225 while (r < n) {
226 bool is_coprime = true;
227 for (uint64_t i = 2; i <= std::sqrt(r); i++) {
228 if (r % i == 0 && n % i == 0) {
229 is_coprime = false;
230 break;
231 }
232 }
233 if (is_coprime) {
234 uint64_t t = 1;
235 for (uint64_t i = 0; i < std::floor(2 * std::log2(n)); i++) {
236 t = (t * r) % n;
237 }
238 if (t == 1) {
239 r++;
240 continue;
241 }
242 bool is_prime = true;
243 for (uint64_t i = 0; i < std::floor(std::log2(n)); i++) {
244 t = (t * t) % n;
245 if (t == 1) {
246 is_prime = false;
247 break;
248 }
249 }
250 if (is_prime) {
251 return true;
252 }
253 }
254 r++;
255 }
256 return false;
257}

259/*
260 * another algorithm capable of finding primes
261 */
262uint64_t mtpk::PrimalityTest::jacobian_number(uint64_t a, uint64_t n) {
263 if (!a)
264 return 0; // (0/n) = 0
265
266 uint64_t ans = 1;
267
268 if (a < 0) {
269 a = -a; // (a/n) = (-a/n)*(-1/n)
270 if (n % 4 == 3)
271 ans = -ans; // (-1/n) = -1 if n = 3 (mod 4)
272 }
273
274 if (a == 1)
275 return ans; // (1/n) = 1
276
277 while (a) {
278 if (a < 0) {
279 a = -a; // (a/n) = (-a/n)*(-1/n)
280 if (n % 4 == 3)
281 ans = -ans; // (-1/n) = -1 if n = 3 (mod 4)
282 }
283
284 while (a % 2 == 0) {
285 a = a / 2;
286 if (n % 8 == 3 || n % 8 == 5)
287 ans = -ans;
288 }
289 std::swap(a, n);
290
291 if (a % 4 == 3 && n % 4 == 3)
292 ans = -ans;
293
294 a = a % n;
295
296 if (a > n / 2)
297 a = a - n;
298 }
299
300 if (n == 1)
301 return ans;
302
303 return 0;
304}
305/*
306 * primality test determining if a number is composite or probably
307 * prime.
308 *
309 * a^(p-1)/2 = (a/p) (mod p)
310 */
311bool mtpk::PrimalityTest::solovoy_strassen(uint64_t p, uint64_t iters) {
312 if (p < 2)
313 return false;
314
315 if (p != 2 && p % 2 == 0)
316 return false;
317
318 for (uint64_t i = 0; i < iters; i++) {
319 // Generate a random number a
320 uint64_t a = rand() % (p - 1) + 1;
321 uint64_t jacobian = (p + jacobian_number(a, p)) % p;
322 uint64_t mod = mod_pow(a, (p - 1) / 2, p);
323
324 if (!jacobian || mod != jacobian)
325 return false;
326 }
327 return true;
328}

330/*
331 * Absolute Fermat Pseudoprimes referred to as Carmichael Numbers
332 * are composite integers satisfying the congruence forumla below
333 * b^n - 1 = b (mod n)
334 */
335bool mtpk::PrimalityTest::carmichael_num(uint64_t n) {
336 for (uint64_t b = 2; b < n; b++) {
337 // If "b" is relatively prime to n
338 if (ba.op_gcd(b, n) == 1)
339 // And pow(b, n-1)%n is not 1, return false.
340 if (mod_pow(b, n - 1, n) != 1)
341 return false;
342 }
343 return true;
344}

380/*
381 * Euler's Totient Function counts the positive numbers up until a
382 * given integer
383 *
384 * Definition: For any positive integer n, Î¦(n) is the number of all
385 * positive integers less than or equal to n that are relatively prime
386 * to n.
387 *
388 * Example: Î¦(1) = 1
389 * Î¦(2) = 1
390 * Î¦(3) = 2
391 * Î¦(4) = 2
392 * Î¦(5) = 4
393 */
394uint64_t mtpk::PrimalityTest::ETF(uint64_t n) {
395 uint64_t result = 1;
396
397 for (uint64_t index = 2; uint64_t(index) < n; index++) {
398 if (ba.op_gcd(index, n) == 1) {
399 result++;

401 }
402
