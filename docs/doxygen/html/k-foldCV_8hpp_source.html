79template <class T>
80KF_CV<T>::KF_CV(const vector<T> &instance_list, int64_t K, uint64_t seed) {
81 this->instance_list = instance_list;
82 shuffle(this->instance_list.begin(), this->instance_list.end(),
83 default_random_engine(seed));
84 N = instance_list.size();
85 this->K = K;
86}

96template <class T> vector<T> KF_CV<T>::get_train_fold(int64_t k) {
97 vector<T> train_fold;
98 for (int64_t i = 0; i < (k * N) / this->K; i++) {
99 train_fold.push_back(instance_list.at(i));
100 }
101 for (int64_t i = ((k + 1) * N) / this->K; i < N; i++) {
102 train_fold.push_back(instance_list.at(i));
103 }
104 return train_fold;
105}
106
114template <class T> vector<T> KF_CV<T>::get_test_fold(int64_t k) {
115 vector<T> test_fold;
116 for (int64_t i = (k * N) / this->K; i < ((k + 1) * N) / this->K; i++) {
117 test_fold.push_back(instance_list.at(i));
118 }
119 return test_fold;
120}
