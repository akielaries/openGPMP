111 template <typename T>
112 void display(std::pair<std::vector<T>, std::vector<std::vector<T>>> data,
113 bool display_all = false) {
114 int num_columns = data.first.size();
115 int num_rows = data.second.size();
116 int num_omitted_rows = 0;
117
118 for (const auto &header : data.first) {
119 std::cout << header << "\t";
120 }
121 std::cout << std::endl;
122
123 int num_elements = data.second.size();
124 if (!display_all && num_elements > 10) {
125 for (int i = 0; i < 5; i++) {
126 for (const auto &value : data.second[i]) {
127 std::cout << value << "\t";
128 }
129 std::cout << std::endl;
130 }
131 num_omitted_rows = num_elements - 10;
132 std::cout << "...\n";
133 std::cout << "[" << num_omitted_rows << " rows omitted]\n";
134 for (int i = num_elements - 5; i < num_elements; i++) {
135 for (const auto &value : data.second[i]) {
136 std::cout << value << "\t";
137 }
138 std::cout << std::endl;
139 }
140 } else {
141 for (const auto &row : data.second) {
142 for (const auto &value : row) {
143 std::cout << value << "\t";
144 }
145 std::cout << std::endl;
146 }
147 }
148
149 std::cout << "[" << num_rows << " rows"
150 << " x " << num_columns << " columns";
151 std::cout << "]\n\n";
152 }
153};
