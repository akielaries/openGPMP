66template <typename T> class Arith {
67 public:
68 T arr_add(T arr[], int64_t n) {
69 T sum = 0;
70
71 for (int64_t i = 0; i < n; i++) {
72 sum += arr[i];
73 }
74 return sum;
75 }
76
77 T arr_sub(T arr[], int64_t n) {
78 T diff = arr[0];
79
80 for (int64_t i = 1; i < n; i++) {
81 diff -= arr[i];
82 }
83 return diff;
84 }
85
86 T arr_mlt(T arr[], int64_t n) {
87 T prod = arr[0];
88
89 for (int64_t i = 0; i < n; i++) {
90 prod *= arr[i];
91 }
92 return prod;
93 }
