109 // Function to calculate and display the best fitting line
110 void gpmp::ml::LinearRegression::best_fit() {
111  if (x_train.empty() || y_train.empty()) {
112  // Check if training data is empty
113  _log_.log(WARNING, "Training data is empty.");
114 
115  if (coeff == 0 && constant == 0) {
116  // If coefficients are not calculated, calculate them
117  calculate_coeffecient();
118  calculate_constant();
119  }
120  // Display the best fitting line equation
121  _log_.log(INFO,
122  "Best fitting line : y = " + std::to_string(coeff) + "x + " +
123  std::to_string(constant));
124  return;
125  }
126 
127  // Calculate the coefficients using the training data
128  long double N = x_train.size();
129  long double sum_xy_train = 0;
130  long double sum_x_train = 0;
131  long double sum_y_train = 0;
132  long double sum_x_square_train = 0;
133 
134  for (size_t i = 0; i < N; i++) {
135  // Calculate sums for the training data
136  sum_xy_train += x_train[i] * y_train[i];
137  sum_x_train += x_train[i];
138  sum_y_train += y_train[i];
139  sum_x_square_train += x_train[i] * x_train[i];
140  }
141 
142  long double numerator = (N * sum_xy_train - sum_x_train * sum_y_train);
143  long double denominator =
144  (N * sum_x_square_train - sum_x_train * sum_x_train);
145 
146  // Calculate the coefficients of the best fitting line
147  coeff = numerator / denominator;
148  constant = (sum_y_train - coeff * sum_x_train) / N;
149  // Display the best fitting line equation
150  _log_.log(INFO,
151  "Best fitting line : y = " + std::to_string(coeff) + "x + " +
152  std::to_string(constant));
153 }

156 void gpmp::ml::LinearRegression::get_input(
157  const std::vector<long double> &x_data,
158  const std::vector<long double> &y_data) {
159  // Clear existing data from x and y vectors
160  x.clear();
161  y.clear();
162  // Initialize LinearRegression class variables
163  sum_xy = 0; /* Set x*y sum */
164  sum_x = 0; /* Set sum of x */
165  sum_y = 0; /* Set sum of y */
166  sum_x_square = 0; /* Set sum of x squares */
167  sum_y_square = 0; /* Set sum of y squares */
168 
169  if (x_data.size() != y_data.size()) {
170  // Check if input vectors have the same size
171  _log_.log(ERROR, "Input vectors must have the same size");
172  return;
173  }
174 
175  for (size_t i = 0; i < x_data.size(); i++) {
176  // Append x and y values to x and y vectors
177  x.push_back(x_data[i]);
178  y.push_back(y_data[i]);
179  // Update sum of (x * y)
180  sum_xy += x_data[i] * y_data[i];
181  // Update sum of x and y
182  sum_x += x_data[i];
183  sum_y += y_data[i];
184  // Update sum of x squares and y squares
185  sum_x_square += x_data[i] * x_data[i];
186  sum_y_square += y_data[i] * y_data[i];
187  }
188 }

194  // Clear any existing data from x and y vectors
195  x.clear();
196  y.clear();

198  sum_x = 0;
199  sum_y = 0;
200  sum_x_square = 0;
201  sum_y_square = 0;
202 
203  // Ensure that columns is not empty and has at least 2 elements
204  if (columns.size() < 2) {
205  _log_.log(ERROR, "Input vectors must have at least 2 column names.");
206  return;
207  }
208 
209  // Find the column indices for the specified column names
210  std::vector<size_t> column_indices;
211  for (const auto &column_name : columns) {
212  bool found = false;
213  for (size_t i = 0; i < data.first.size(); ++i) {
214  if (data.first[i] == column_name) {
215  column_indices.push_back(i);
216  found = true;
217  break;
218  }
219  }
220  if (!found) {
221  _log_.log(ERROR,
222  "Column '" + column_name +
223  "' not found in DataTableStr.");
224  return;
225  }
226  }
227 
228  for (const auto &row : data.second) {
229  try {
230  long double xi = std::stold(row[column_indices[0]]);
231  long double yi = std::stold(row[column_indices[1]]);
232  // Append x and y values to x and y vectors
233  x.push_back(xi);
234  y.push_back(yi);
235  // Update sum of (x * y)
236  sum_xy += xi * yi;
237  // Update sum of x and y
238  sum_x += xi;
239  sum_y += yi;
240  // Update sum of x squares and y squares
241  sum_x_square += xi * xi;
242  sum_y_square += yi * yi;
243  } catch (const std::exception &e) {
244  // Handle parsing errors here
245  _log_.log(ERROR, "Error parsing data: " + std::string(e.what()));
246  continue;
247  }
248  }

277 // Function to split data into training and testing sets
278 void gpmp::ml::LinearRegression::split_data(double test_size,
279  unsigned int seed,
280  bool shuffle) {
281  if (x.empty() || y.empty()) {
282  _log_.log(ERROR, "Training data is empty.");
283  return;
284  }
285 
286  if (test_size <= 0 || test_size >= 1) {
287  _log_.log(ERROR, "Invalid `test_size`. Must be between 0 - 1.");
288  return;
289  }
290 
291  size_t data_size = x.size();
292  size_t test_data_size = static_cast<size_t>(test_size * data_size);
293 
294  // Shuffle the data randomly if specified
295  if (shuffle == true) {
296  // Create vector of indices
297  std::vector<size_t> indices(data_size);
298  // Fill vector sequentially w/ `iota`
299  std::iota(indices.begin(), indices.end(), 0);
300  // Randomly shuffle vector indices from start to end
301  std::shuffle(indices.begin(),
302  indices.end(),
303  std::default_random_engine(seed));
304 
305  // Clear training and testing data vectors
306  x_train.clear();
307  y_train.clear();
308  x_test.clear();
309  y_test.clear();
310 
311  // Split the data into training and testing sets based on shuffled
312  // indices
313  for (size_t i = 0; i < data_size; ++i) {
314  if (i < test_data_size) {
315  // Append x test vector with shuffled x value
316  x_test.push_back(x[indices[i]]);
317  y_test.push_back(y[indices[i]]);
318  } else {
319  // Append x training vector with suffled x value
320  x_train.push_back(x[indices[i]]);
321  y_train.push_back(y[indices[i]]);
322  }
323  }
324  } else {
325  // Without shuffling, split the data without changing its order by
326  // assigning the first training element to the training set
327  x_train.assign(x.begin(), x.begin() + data_size - test_data_size);
328  y_train.assign(y.begin(), y.begin() + data_size - test_data_size);
329  // Assign the 1+ testing elements to the testing test
330  x_test.assign(x.begin() + data_size - test_data_size, x.end());
331  y_test.assign(y.begin() + data_size - test_data_size, y.end());
332  }
333 }

408 // Find the Mean Squared Error (MSE) of the given dataset
409 long double
410 gpmp::ml::LinearRegression::mse(const std::vector<long double> &x_data,
411  const std::vector<long double> &y_data) const {
412  // Check if input vectors have the same size
413  if (x_data.size() != y_data.size()) {
414  return -1; // Return an error value
415  }
416 
417  long double mse = 0.0;
418  int64_t n = x_data.size();
419 
420  // Iterate through each data point
421  for (int64_t i = 0; i < n; i++) {
422  // Calculate the predicted value using the linear regression model
423  long double predicted = predict(x_data[i]);
424  // Calculate the error (difference between predicted and actual values)
425  long double error = predicted - y_data[i];
426  // Accumulate the squared error
427  mse += error * error;
428  }
429 
430  // Calculate the Mean Squared Error by dividing the accumulated squared
431  // error by the number of data points
432  return mse / static_cast<long double>(n);
433 }

435 // Determine an R^2 score value
436 long double gpmp::ml::LinearRegression::r_sqrd(
437  const std::vector<long double> &x_data,
438  const std::vector<long double> &y_data) const {
439  // Check if input vectors have the same size
440  if (x_data.size() != y_data.size()) {
441  _log_.log(ERROR, "Input vectors must have the same size.");
442  return -1; // Return an error value
443  }
444 
445  long double total_sum_of_squares = 0.0;
446  long double sum_of_squared_errors = 0.0;
447  int64_t n = x_data.size();
448 
449  long double y_mean = 0.0;
450  for (int64_t i = 0; i < n; i++) {
451  // Calculate the mean of the dependent variable (Y)
452  y_mean += y_data[i];
453  }
454  y_mean /= static_cast<long double>(n);
455 
456  // Iterate through each data point
457  for (int64_t i = 0; i < n; i++) {
458  // Calculate the predicted value using the linear regression model
459  long double predicted = predict(x_data[i]);
460  // Calculate the error (difference between predicted and actual values)
461  long double error = predicted - y_data[i];
462  // Calculate the total sum of squares and sum of squared errors
463  total_sum_of_squares += (y_data[i] - y_mean) * (y_data[i] - y_mean);
464  sum_of_squared_errors += error * error;
465  }
466 
467  // Calculate the R-squared value using the formula 1 - (SSE / SST)
468  return 1.0 - (sum_of_squared_errors / total_sum_of_squares);
469 }
