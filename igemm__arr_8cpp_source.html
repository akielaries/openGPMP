146 // micro kernel that multiplies panels from A and B
147 void gpmp::linalg::IGEMM::igemm_micro_kernel(int kc,
148  int alpha,
149  const int *A,
150  const int *B,
151  int beta,
152  int *C,
153  int incRowC,
154  int incColC) {
155  int AB[BLOCK_SZ_MR * BLOCK_SZ_NR];
156 
157  int i, j, l;
158 
159  // Compute AB = A*B
160  for (l = 0; l < BLOCK_SZ_MR * BLOCK_SZ_NR; ++l) {
161  AB[l] = 0;
162  }
163  for (l = 0; l < kc; ++l) {
164  for (j = 0; j < BLOCK_SZ_NR; ++j) {
165  for (i = 0; i < BLOCK_SZ_MR; ++i) {
166  AB[i + j * BLOCK_SZ_MR] += A[i] * B[j];
167  }
168  }
169  A += BLOCK_SZ_MR;
170  B += BLOCK_SZ_NR;
171  }
172 
173  // Update C <- beta*C
174  if (beta == 0) {
175  for (j = 0; j < BLOCK_SZ_NR; ++j) {
176  for (i = 0; i < BLOCK_SZ_MR; ++i) {
177  C[i * incRowC + j * incColC] = 0;
178  }
179  }
180  } else if (beta != 1) {
181  for (j = 0; j < BLOCK_SZ_NR; ++j) {
182  for (i = 0; i < BLOCK_SZ_MR; ++i) {
183  C[i * incRowC + j * incColC] *= beta;
184  }
185  }
186  }
187 
188  // Update C <- C + alpha*AB (note: the case alpha==0 was already treated
189  // in
190  // the above layer igemm_nn)
191  if (alpha == 1) {
192  for (j = 0; j < BLOCK_SZ_NR; ++j) {
193  for (i = 0; i < BLOCK_SZ_MR; ++i) {
194  C[i * incRowC + j * incColC] += AB[i + j * BLOCK_SZ_MR];
195  }
196  }
197  }
198 
199  else {
200  for (j = 0; j < BLOCK_SZ_NR; ++j) {
201  for (i = 0; i < BLOCK_SZ_MR; ++i) {
202  C[i * incRowC + j * incColC] += alpha * AB[i + j * BLOCK_SZ_MR];
203  }
204  }
205  }
206 }
