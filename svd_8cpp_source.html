61 void gpmp::linalg::SVD::computeSVD(std::vector<std::vector<double>> &matrix) {
62  size_t m = matrix.size();
63  size_t n = matrix[0].size();
64 
65  singularValues_.resize(std::min(m, n));
66  leftSingularVectors_ =
67  std::vector<std::vector<double>>(m, std::vector<double>(m, 0.0));
68  rightSingularVectors_ =
69  std::vector<std::vector<double>>(n, std::vector<double>(n, 0.0));
70 
71  std::vector<double> superdiagonal(n - 1, 0.0);
72 
73  bidiagonalize(const_cast<std::vector<std::vector<double>> &>(matrix),
74  singularValues_,
75  superdiagonal);
76 
77  // left/right singular vectors as identity matrices
78  for (size_t i = 0; i < m; ++i) {
79  leftSingularVectors_[i][i] = 1.0;
80  }
81

83  rightSingularVectors_[i][i] = 1.0;
84  }

87 void gpmp::linalg::SVD::bidiagonalize(std::vector<std::vector<double>> &matrix,
88  std::vector<double> &diagonal,
89  std::vector<double> &superdiagonal) {
90  size_t m = matrix.size();
91  size_t n = matrix[0].size();
92 
93  diagonal.resize(std::min(m, n));
94  superdiagonal.resize(std::min(m, n) - 1);
95 
96  for (size_t k = 0; k < std::min(m, n); ++k) {
97  double alpha = 0.0;
98  for (size_t i = k; i < m; ++i) {
99  alpha += matrix[i][k] * matrix[i][k];
100  }
101  alpha = (matrix[k][k] > 0) ? -std::sqrt(alpha) : std::sqrt(alpha);
102 
103  double beta = alpha * (alpha - matrix[k][k]);
104  diagonal[k] = -alpha;
105 
106  // make a copy of the current column of the matrix
107  std::vector<double> columnCopy(m - k);
108  for (size_t i = k; i < m; ++i) {
109  columnCopy[i - k] = matrix[i][k];
110  }
111 
112  computeHouseholderReflection(columnCopy, superdiagonal, beta);
113  applyHouseholder(matrix, superdiagonal, beta, k, k + 1);
114  }
115 }

117 void gpmp::linalg::SVD::computeHouseholderReflection(
118  const std::vector<double> &x,
119  std::vector<double> &v,
120  double &beta) {
121  size_t n = x.size();
122  double sigma = 0.0;
123 
124  for (size_t i = 1; i < n; ++i) {
125  sigma += x[i] * x[i];
126  }
127 
128  v.resize(n);
129  v[0] = 1.0;
130  for (size_t i = 1; i < n; ++i) {
131  v[i] = x[i] / sigma;
132  }
133 
134  beta = 2.0 / (sigma + 1.0);
135 }

137 void gpmp::linalg::SVD::applyHouseholder(
138  std::vector<std::vector<double>> &matrix,
139  const std::vector<double> &v,
140  double beta,
141  size_t fromRow,
142  size_t fromCol) {
143  size_t m = matrix.size();
144  size_t n = matrix[0].size();
145 
146  for (size_t i = fromRow; i < m; ++i) {
147  double dotProduct = 0.0;
148  for (size_t j = fromCol; j < n; ++j) {
149  dotProduct += matrix[i][j] * v[j - fromCol];
150  }
151 
152  dotProduct *= beta;
153 
154  for (size_t j = fromCol; j < n; ++j) {
155  matrix[i][j] -= dotProduct * v[j - fromCol];
156  }
157  }
158 }
