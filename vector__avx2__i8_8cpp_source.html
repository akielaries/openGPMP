62 template <typename T>
63 void gpmp::linalg::vector_add_i8(const T *data1,
64  const T *data2,
65  T *result_data,
66  size_t size) {
67  size_t i = 0;
68  if (size > 64) {
69  for (; i < size - 31; i += 32) {
70  __m256i a = _mm256_loadu_si256(
71  reinterpret_cast<const __m256i *>(data1 + i));
72  __m256i b = _mm256_loadu_si256(
73  reinterpret_cast<const __m256i *>(data2 + i));
74  __m256i c = _mm256_add_epi8(a, b);
75  _mm256_storeu_si256(reinterpret_cast<__m256i *>(result_data + i),
76  c);
77  }
78  }
79  for (; i < size; ++i) {
80  result_data[i] = data1[i] + data2[i];
81  }
82 }

139 template <typename T>
140 void gpmp::linalg::scalar_mult_i8(const T *data,
141  int scalar,
142  T *result_data,
143  size_t size) {
144  size_t i = 0;
145 
146  if (size > 64) {
147  __m256i scalar_vec = _mm256_set1_epi16(scalar);
148 
149  for (; i < size - 31; i += 32) {
150  // Load 32 elements from vec
151  __m256i a =
152  _mm256_loadu_si256(reinterpret_cast<const __m256i *>(data + i));
153 
154  // Split the 16-bit values into two 8-bit halves
155  __m256i a_low = _mm256_unpacklo_epi8(a, _mm256_setzero_si256());
156  __m256i a_high = _mm256_unpackhi_epi8(a, _mm256_setzero_si256());
157 
158  // Perform vectorized multiplication of 16-bit integers
159  __m256i c_low = _mm256_mullo_epi16(a_low, scalar_vec);
160  __m256i c_high = _mm256_mullo_epi16(a_high, scalar_vec);
161 
162  // Pack the 16-bit integers back to 8-bit integers
163  __m256i c = _mm256_packus_epi16(c_low, c_high);
164 
165  // Store the result back to result vector
166  _mm256_storeu_si256(reinterpret_cast<__m256i *>(result_data + i),
167  c);
168  }
169  }
170  for (; i < size; ++i) {
171  result_data[i] = data[i] * scalar;
172  }
173 }

191 template <typename T>
192 T gpmp::linalg::dot_product_i8(const T *vec1, const T *vec2, size_t size) {
193  int result = 0;
194  if (size > 32) {
195  size_t i = 0;
196  // Perform vectorized multiplication and addition as long as there are
197  // at least 16 elements remaining
198  for (; i < size - 15; i += 16) {
199  // Load 16 elements from vec1 and vec2, unpacking them to 16-bit
200  // integers
201  __m256i a = _mm256_cvtepu8_epi16(
202  _mm_loadu_si128(reinterpret_cast<const __m128i *>(vec1 + i)));
203  __m256i b = _mm256_cvtepu8_epi16(
204  _mm_loadu_si128(reinterpret_cast<const __m128i *>(vec2 + i)));
205 
206  // Perform vectorized multiplication and addition
207  __m256i c = _mm256_mullo_epi16(a, b);
208  __m256i sum = _mm256_hadd_epi16(c, c);
209  sum = _mm256_hadd_epi16(sum, sum);
210  sum = _mm256_hadd_epi16(sum, sum);
211 
212  // Accumulate the result
213  result += _mm256_extract_epi16(sum, 0);
214  result += _mm256_extract_epi16(sum, 8);
215  }
216 
217  // Perform standard dot product on the remaining elements
218  for (; i < size; ++i) {
219  result += vec1[i] * vec2[i];
220  }
221  } else {
222  for (size_t i = 0; i < size; ++i) {
223  result += vec1[i] * vec2[i];
224  }
225  }
226 
227  return result;
228 }
