68 gpmp::Differential gpmp::Differential::power_rule() const {
69  gpmp::Differential result;
70  for (const auto &term : terms) {
71  if (term.exponent > 0) {
72  double newCoefficient = term.coefficient * term.exponent;
73  int newExponent = term.exponent - 1;

75  }
76  }
77  return result;

81 gpmp::Differential::chain_rule(const gpmp::Differential &inner) const {
82  gpmp::Differential result;
83 
84  for (const auto &outerTerm : terms) {
85  // Apply the chain rule to each term of the outer function
86  gpmp::Differential innerDerivative = inner.power_rule();
87 
88  // Multiply each term of innerDerivative by the coefficient of the outer
89  // term
90  for (auto &innerTerm : innerDerivative.terms) {
91  innerTerm.coefficient *= outerTerm.coefficient;
92  }
93 
94  // Multiply the inner derivative by the derivative of the outer term
95  for (const auto &innerTerm : innerDerivative.terms) {
96  double newCoefficient = innerTerm.coefficient;
97  int newExponent = outerTerm.exponent + innerTerm.exponent;
98  result.add_term(newCoefficient, newExponent);
99  }
100  }
101 
102  return result;
103 }

131 double gpmp::Differential::limit_at_infinity() const {
132  // Calculate the limit as x approaches infinity using the highest exponent
133  // term
134  if (!terms.empty()) {
135  const auto &highest_term =
136  *std::max_element(terms.begin(),
137  terms.end(),
138  [](const auto &a, const auto &b) {
139  return a.exponent < b.exponent;
140  });
141 
142  if (highest_term.exponent > 0) {
143  // If the highest term has a positive exponent, the limit is
144  // infinity or negative infinity
145  return (highest_term.coefficient > 0)
146  ? std::numeric_limits<double>::infinity()
147  : -std::numeric_limits<double>::infinity();
148  } else {
149  // If the highest term has a zero exponent, the limit is the
150  // constant term
151  return highest_term.coefficient;
152  }
153  }
154 
155  // If the differential is empty, the limit is undefined
156  return std::numeric_limits<double>::quiet_NaN();
157 }
