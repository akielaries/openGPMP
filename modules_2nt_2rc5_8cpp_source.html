52  for (uint32_t i = 1; i <= r; ++i) {
53  A = rotl((A ^ B), B) + S[2 * i];

59  for (uint32_t i = r; i > 0; --i) {
60  B = rotr(B - S[2 * i + 1], A) ^ A;
61  A = rotr(A - S[2 * i], B) ^ B;
62  }
63  B -= S[1];
64  A -= S[0];
65 }
