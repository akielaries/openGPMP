69 std::vector<std::pair<int, double>> gpmp::linalg::sigproc::gaussian_filter(
70  const std::vector<std::pair<int, double>> &signal,
71  double sigma) {
72  std::vector<std::pair<int, double>> filtered_signal;
73 
74  // Iterate over each element in the signal
75  for (size_t i = 0; i < signal.size(); ++i) {
76  double filtered_value = 0.0;
77  double sum_weights = 0.0;
78 
79  // Compute weighted average of neighboring values
80  for (size_t j = 0; j < signal.size(); ++j) {
81  double distance = abs(signal[i].first - signal[j].first);
82  double weight = gaussian(distance, 0, sigma);
83  filtered_value += weight * signal[j].second;
84  sum_weights += weight;
85  }
86 
87  // Normalize filtered value
88  filtered_value /= sum_weights;
89 
90  // Add filtered value to filtered signal
91  filtered_signal.push_back({signal[i].first, filtered_value});
92  }
93 
94  return filtered_signal;
95 }

97 std::vector<std::pair<int, double>> gpmp::linalg::sigproc::butterworth_lpf(
98  const std::vector<std::pair<int, double>> &signal,
99  double cutoff_frequency,
100  double sampling_frequency,
101  int order) {
102  // Calculate the normalized cutoff frequency
103  double omega_c = 2.0 * M_PI * cutoff_frequency / sampling_frequency;
104 
105  // Calculate the poles of the Butterworth filter
106  std::vector<std::complex<double>> poles;
107  for (int k = 0; k < order; ++k) {
108  double real_part = -sin((2 * k + 1) * M_PI / (2 * order));
109  double imag_part = cos((2 * k + 1) * M_PI / (2 * order));
110  poles.push_back(std::complex<double>(real_part, imag_part));
111  }
112 
113  // Filter the signal using the Butterworth filter
114  std::vector<std::pair<int, double>> filtered_signal;
115  for (size_t i = 0; i < signal.size(); ++i) {
116  double t = signal[i].first; // Time index
117  double x = signal[i].second; // Signal value
118 
119  // Compute the filtered value at time t
120  double filtered_value = 0.0;
121  for (const auto &pole : poles) {
122  std::complex<double> z(cos(omega_c), sin(omega_c));
123  std::complex<double> num = 1.0;
124  std::complex<double> denom = z - pole;
125  for (int j = 1; j < order; ++j) {
126  num *= z;
127  denom *= (z - poles[j]);
128  }
129  filtered_value += std::real(num / denom);
130  }
131  filtered_value *= x;
132 
133  // Add the filtered value to the filtered signal
134  filtered_signal.push_back({t, filtered_value});
135  }
136 
137  return filtered_signal;
138 }

140 std::vector<std::pair<int, double>> gpmp::linalg::sigproc::chebyshev_lpf(
141  const std::vector<std::pair<int, double>> &signal,
142  double cutoff_frequency,
143  double sampling_frequency,
144  int order,
145  double ripple) {
146 
147  // Calculate the normalized cutoff frequency
148  double omega_c = 2.0 * M_PI * cutoff_frequency / sampling_frequency;
149 
150  // Calculate the ripple factor (epsilon)
151  double epsilon = sqrt(pow(10, ripple / 10) - 1);
152 
153  // Calculate the poles of the Chebyshev filter
154  std::vector<std::complex<double>> poles;
155  for (int k = 0; k < order; ++k) {
156  double real_part = -sinh(asinh(1 / epsilon) / order) *
157  sin((2 * k + 1) * M_PI / (2 * order));
158  double imag_part = cosh(asinh(1 / epsilon) / order) *
159  cos((2 * k + 1) * M_PI / (2 * order));
160  poles.push_back(std::complex<double>(real_part, imag_part));
161  }
162 
163  // Filter the signal using the Chebyshev filter
164  std::vector<std::pair<int, double>> filtered_signal;
165  for (size_t i = 0; i < signal.size(); ++i) {
166  double t = signal[i].first; // Time index
167  double x = signal[i].second; // Signal value
168 
169  // Compute the filtered value at time t
170  double filtered_value = 0.0;
171  for (const auto &pole : poles) {
172  std::complex<double> z(cos(omega_c), sin(omega_c));
173  std::complex<double> num = 1.0;
174  std::complex<double> denom = z - pole;
175  for (int j = 0; j < order; ++j) {
176  if (j != i) {
177  num *= z;
178  denom *= (z - poles[j]);
179  }
180  }
181  filtered_value += std::real(num / denom);
182  }
183  filtered_value *= x;
184 
185  // Add the filtered value to the filtered signal
186  filtered_signal.push_back({t, filtered_value});
187  }
188 
189  return filtered_signal;
190 }

192 std::vector<std::pair<int, double>> gpmp::linalg::sigproc::bessel_lpf(
193  const std::vector<std::pair<int, double>> &signal,
194  double cutoff_frequency,
195  double sampling_frequency,
196  int order) {
197  // Calculate the normalized cutoff frequency
198  double omega_c = 2.0 * M_PI * cutoff_frequency / sampling_frequency;
199 
200  // Calculate the poles of the Bessel filter
201  std::vector<std::complex<double>> poles;
202  for (int k = 1; k <= order; ++k) {
203  double real_part =
204  -sin(M_PI * (2 * k + 1) / (2 * order)) * sinh(log(2) / (2 * order));
205  double imag_part =
206  cos(M_PI * (2 * k + 1) / (2 * order)) * cosh(log(2) / (2 * order));
207  poles.push_back(std::complex<double>(real_part, imag_part));
208  }
209 
210  // Filter the signal using the Bessel filter
211  std::vector<std::pair<int, double>> filtered_signal;
212  for (size_t i = 0; i < signal.size(); ++i) {
213  double t = signal[i].first; // Time index
214  double x = signal[i].second; // Signal value
215 
216  // Compute the filtered value at time t
217  double filtered_value = 0.0;
218  for (const auto &pole : poles) {
219  std::complex<double> z(cos(omega_c), sin(omega_c));
220  std::complex<double> num = 1.0;
221  std::complex<double> denom = z - pole;
222  for (const auto &p : poles) {
223  if (p != pole) {
224  num *= (z - p);
225  denom *= (pole - p);
226  }
227  }
228  filtered_value += std::real(num / denom);
229  }
230  filtered_value *= x;
231 
232  // Add the filtered value to the filtered signal
233  filtered_signal.push_back({t, filtered_value});
234  }
235 
236  return filtered_signal;
237 }

337 std::vector<double> gpmp::linalg::sigproc::compute_dct(size_t M) const {
338  size_t N = signal_.size();
339  std::vector<double> result(M, 0.0);
340 
341  for (size_t m = 0; m < M; ++m) {
342  double sum = 0.0;
343  double coef = (m == 0) ? 1.0 / std::sqrt(2.0) : 1.0;
344 
345  for (size_t n = 0; n < N; ++n) {
346  double theta = (M_PI * m * (2.0 * n + 1)) / (2.0 * N);
347  sum += signal_[n] * std::cos(theta);
348  }
349 
350  result[m] = coef * sum;
351  }
352 
353  return result;
354 }

482  const std::vector<double> &analog_system_coefficients,
483  double sampling_rate) {
484  // Design a discrete-time system using impulse invariance method
485  size_t order = analog_system_coefficients.size() - 1;
486  std::vector<double> discrete_system_coefficients(order + 1, 0.0);
487 
488  // Impulse invariance transformation

609 std::vector<double> gpmp::linalg::sigproc::apply_anti_aliasing_filter(
610  const std::vector<double> &input_signal,
611  double sampling_rate,
612  double cutoff_frequency) {
613  // Apply a simple low-pass filter as an anti-aliasing measure.
614  // Note: This is a basic example; actual anti-aliasing filters may require
615  // more sophisticated designs.
616 
617  size_t num_samples = input_signal.size();
618  std::vector<double> anti_aliased_signal(num_samples);
619 
620  // Design a simple first-order low-pass filter.
621  double RC = 1.0 / (2.0 * M_PI * cutoff_frequency);
622  double alpha = 1.0 / (1.0 + RC * sampling_rate);
623 
624  // Apply the low-pass filter.
625  anti_aliased_signal[0] = input_signal[0];
626  for (size_t i = 1; i < num_samples; ++i) {
627  anti_aliased_signal[i] = alpha * input_signal[i] +
628  (1.0 - alpha) * anti_aliased_signal[i - 1];
629  }
630 
631  return anti_aliased_signal;
632 }
