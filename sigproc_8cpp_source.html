50 double SigProc::stdev() const {
51  double mn = mean();

53 
54  for (const auto &value : signal_) {
55  double diff = value - mn;

188 std::vector<double> SigProc::compute_dct(size_t M) const {
189  size_t N = signal_.size();
190  std::vector<double> result(M, 0.0);
191 
192  for (size_t m = 0; m < M; ++m) {
193  double sum = 0.0;
194  double coef = (m == 0) ? 1.0 / std::sqrt(2.0) : 1.0;
195 
196  for (size_t n = 0; n < N; ++n) {
197  double theta = (M_PI * m * (2.0 * n + 1)) / (2.0 * N);
198  sum += signal_[n] * std::cos(theta);
199  }
200 
201  result[m] = coef * sum;
202  }
203 
204  return result;
205 }

332 std::vector<double> SigProc::design_discrete_system(
333  const std::vector<double> &analog_system_coefficients,
334  double sampling_rate) {
335  // Design a discrete-time system using impulse invariance method
336  size_t order = analog_system_coefficients.size() - 1;
337  std::vector<double> discrete_system_coefficients(order + 1, 0.0);
338 
339  // Impulse invariance transformation
340  for (size_t k = 0; k <= order; ++k) {
341  double ak = analog_system_coefficients[k];
342  for (size_t n = 0; n <= order; ++n) {
343  discrete_system_coefficients[n] +=
344  ak * std::sin(2 * M_PI * n * sampling_rate / 2.0) / (M_PI * n);
345  }
346  }
347 
348  return discrete_system_coefficients;
349 }

352 SigProc::impulse_invariance_transform(const std::vector<double> &analog_signal,
353  double analog_sampling_rate,
354  double discrete_sampling_rate) {
355  size_t M = analog_signal.size();
356  std::vector<double> discrete_signal(M, 0.0);
357 
358  // Impulse invariance transformation
359  for (size_t m = 0; m < M; ++m) {
360  double t = m / analog_sampling_rate;
361  double sum = 0.0;
362 
363  for (size_t n = 0; n < M; ++n) {
364  sum += analog_signal[n] *
365  std::sin(M_PI * discrete_sampling_rate *
366  (t - n / discrete_sampling_rate)) /
367  (M_PI * (t - n / discrete_sampling_rate));
368  }
369 
370  discrete_signal[m] = sum;
371  }
372 
373  return discrete_signal;
