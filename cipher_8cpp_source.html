64 std::string mtpk::Cipher::keyword_encode(std::string key) {
65  std::string encoded = "";
66  // This array represents the 26 letters of alphabets
67  bool arr[26] = {0};
68 
69  // This loop inserts the keyword at the start of the encoded
70  // string
71  for (int64_t i = 0; uint64_t(i) < key.size(); i++) {
72  if (key[i] >= 'A' && key[i] <= 'Z') {
73  /*
74  * To check whether the character is inserted earlier
75  * in the encoded string or not
76  */
77  if (arr[key[i] - 65] == 0) {
78  encoded += key[i];
79  arr[key[i] - 65] = 1;
80  }
81  } else if (key[i] >= 'a' && key[i] <= 'z') {
82  if (arr[key[i] - 97] == 0) {
83  encoded += key[i] - 32;
84  arr[key[i] - 97] = 1;
85  }
86  }
87  }
88 
89  // This loop inserts the remaining characters in the encoded
90  // string.
91  for (int64_t i = 0; i < 26; i++) {
92  if (arr[i] == 0) {
93  arr[i] = 1;
94  encoded += char(i + 65);
95  }
96  }
97  return encoded;
98 }
