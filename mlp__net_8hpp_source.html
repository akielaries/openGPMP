162  long double lr = .001)

164  // traverse the elements

188  auto prop_forwards(Matrix<T> x) {
189  assert(get<0>(x.shape) == layer_units[0] && get<1>(x.shape));
190  // input = to previously declared acitvations method
191  activations[0] = x;
192  Matrix prev(x);
193 
194  // traverse layer units
195  for (uint64_t i = 0; i < layer_units.size() - 1; ++i) {
196  Matrix y = wt_mtx[i].mult(prev);
197  y = y + bias_vectors[i];
198  y = y.apply_function(sigmoid_activ);
199  activations[i + 1] = y;

212  void prop_backwards(Matrix<T> target) {
213  assert(get<0>(target.shape) == layer_units.back());
214  // calculate the error, target - ouput
215  auto y = target;
216  auto y_hat = activations.back();
217  auto err = target - y_hat;
218  // back propagate the error calculated from output to input
219  // and step the weights
220  for (int64_t i = wt_mtx.size() - 1; i >= 0; --i) {
221  // calculate errors for previous layer
222  auto wt = wt_mtx[i].T();
223  auto prior_errs = wt.mult(err);
224 
225  auto outputs_d =
226  activations[i + 1].apply_function(sigmoid_deriv);
227  auto gradients = err.hadamard(outputs_d);
228  gradients = gradients.scalar_mult(lr);
229  auto trans_a = activations[i].T();
230  auto gradients_wt = gradients.mult(trans_a);

234  bias_vectors[i] = bias_vectors[i].add(gradients);
