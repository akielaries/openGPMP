193  explicit SecondaryMLP(std::vector<size_t> layer_units,
194  long double lr = .001)
195  : layer_units(layer_units), wt_mtx(), bias_vectors(), lr(lr) {
196  // traverse the elements
197  for (size_t i = 0; i < layer_units.size() - 1; ++i) {
198  // size of inputs
199  size_t inputs{layer_units[i]};
200  // size of outputs
201  size_t outputs{layer_units[i + 1]};
202 
203  // set to random Guassian Noise related values
204  // weights
205  auto gauss_wt = gpmp::linalg::mtx<T>::randn(outputs, inputs);
206  wt_mtx.push_back(gauss_wt);
207  // biases
208  auto bias_wt = gpmp::linalg::mtx<T>::randn(outputs, 1);
209  bias_vectors.push_back(bias_wt);
210  // activation function
211  activations.resize(layer_units.size());
212  }
213  }
220  auto prop_forwards(gpmp::linalg::Matrix<T> x) {
221  assert(get<0>(x.shape) == layer_units[0] && get<1>(x.shape));
222  // input = to previously declared acitvations method
223  activations[0] = x;
224  gpmp::linalg::Matrix prev(x);
225 
226  // traverse layer units
227  for (uint64_t i = 0; i < layer_units.size() - 1; ++i) {
228  gpmp::linalg::Matrix y = wt_mtx[i].mult(prev);
229  y = y + bias_vectors[i];
230  y = y.apply_function(&SecondaryMLP::sigmoid_activ);
231  activations[i + 1] = y;
232  prev = y;
233  }
234  return prev;
235  }
244  void prop_backwards(gpmp::linalg::Matrix<T> target) {
245  assert(get<0>(target.shape) == layer_units.back());
246  // calculate the error, target - ouput
247  auto y = target;
248  auto y_hat = activations.back();
249  auto err = target - y_hat;
250  // back propagate the error calculated from output to input
251  // and step the weights
252  for (int64_t i = wt_mtx.size() - 1; i >= 0; --i) {
253  // calculate errors for previous layer
254  auto wt = wt_mtx[i].T();
255  auto prior_errs = wt.mult(err);
256  auto outputs_d =
257  activations[i + 1].apply_function(&SecondaryMLP::sigmoid_deriv);
258  auto gradients = err.hadamard(outputs_d);
259  gradients = gradients.scalar_mult(lr);
260  auto trans_a = activations[i].T();
261  auto gradients_wt = gradients.mult(trans_a);
262 
263  // adjust the networks weights based on propagation
264  // technique
265  bias_vectors[i] = bias_vectors[i].add(gradients);
266  wt_mtx[i] = wt_mtx[i].add(gradients_wt);
267  err = prior_errs;
268  }
269  }
270 };
