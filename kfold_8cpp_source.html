46 void gpmp::ml::Kfold::split_indices(int num_instances) {
47  if (k <= 1 || num_instances <= 0) {
48  throw std::invalid_argument(
49  "Invalid value of k or number of instances.");
50  }
51 
52  // Initialize fold_indices with k empty vectors
53  fold_indices.clear();
54  fold_indices.resize(k);
55 
56  // Assign indices to each fold
57  int fold_size = num_instances / k;
58  int remainder = num_instances % k;
59  int start_index = 0;
60  for (int i = 0; i < k; ++i) {
61  int fold_length = fold_size + (i < remainder ? 1 : 0);
62  std::vector<int> indices(fold_length);
63  std::iota(indices.begin(), indices.end(), start_index);
64  fold_indices[i] = indices;
65  start_index += fold_length;
66  }
67 }

71  if (fold_indices.empty()) {
72  throw std::logic_error("Indices not split. Call split_indices before "
73  "performing cross-validation.");
74  }
75 
76  for (int i = 0; i < k; ++i) {
77  // Use fold i as the validation set, train on all other folds
78  std::vector<int> validation_indices = fold_indices[i];
79  for (int j = 0; j < k; ++j) {
80  if (j != i) {
81  for (int index : fold_indices[j]) {
82  train_and_test_func(j, index);
83  }
84  }
85  }
