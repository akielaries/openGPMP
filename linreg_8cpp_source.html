59 // Function that calculate the coefficient/slope of the best fitting
60 // line
61 void gpmp::ml::LinearRegression::calculate_coeffecient() {
62  long double N = x.size();
63  long double numerator = (N * sum_xy - sum_x * sum_y);
64  long double denominator = (N * sum_x_square - sum_x * sum_x);
65  coeff = numerator / denominator;
66 }

68 /*
69  * Member function that will calculate the constant term of the best
70  * fitting line
71  */
72 void gpmp::ml::LinearRegression::calculate_constant() {
73  long double N = x.size();
74  long double numerator = (sum_y * sum_x_square - sum_x * sum_xy);
75  long double denominator = (N * sum_x_square - sum_x * sum_x);
76  constant = numerator / denominator;
