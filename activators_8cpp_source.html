55 gpmp::ml::Activation::softmax(const std::vector<double> &inputs) {
56  std::vector<double> result;
57  double sum_exp = 0.0;
58  for (double input : inputs) {
59  sum_exp += exp(input);
60  }
61  for (double input : inputs) {
62  result.push_back(exp(input) / sum_exp);
63  }

67 std::vector<std::vector<double>>
68 gpmp::ml::Activation::softmax_derivative(const std::vector<double> &inputs) {
69  std::vector<std::vector<double>> result(
70  inputs.size(),
71  std::vector<double>(inputs.size(), 0.0));
72  std::vector<double> softmax_values = softmax(inputs);
73  for (size_t i = 0; i < inputs.size(); ++i) {
74  for (size_t j = 0; j < inputs.size(); ++j) {
75  if (i == j) {
76  result[i][j] = softmax_values[i] * (1.0 - softmax_values[i]);
77  } else {
78  result[i][j] = -softmax_values[i] * softmax_values[j];
79  }
80  }
81  }
82  return result;
83 }

112 double gpmp::ml::Activation::gelu_derivative(double z) {
113  double cdf =
114  0.5 * (1.0 + tanh(sqrt(2.0 / M_PI) * (0.044715 * z * z * z + 3 * z)));
115  double pdf = exp(-0.5 * z * z) / sqrt(2.0 * M_PI);
116  return 0.5 * (1.0 + cdf +
117  z * pdf * (1.0 / M_PI) *
118  (0.5 * (1.0 + tanh(sqrt(2.0 / M_PI) *
119  (0.044715 * z * z * z + 3 * z)))) +
120  (1.0 / M_PI) * (1.0 - cdf * cdf));
121 }
