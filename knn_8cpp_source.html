56 int gpmp::ml::KNN::predict(const std::vector<double> &input_vector, int k) {
57  if (training_data.empty() || labels.empty()) {
58  throw std::logic_error(
59  "Model not trained. Call train() before predict.");
60  }
61 
62  if (input_vector.size() != training_data[0].size()) {
63  throw std::invalid_argument("Invalid input vector size.");
64  }
65 
66  if (k <= 0 || k > training_data.size()) {
67  throw std::invalid_argument("Invalid value of k.");
68  }
69 
70  // Calculate distances and store index-label pairs
71  std::vector<std::pair<double, int>> distances;
72  for (size_t i = 0; i < training_data.size(); ++i) {
73  double distance =
74  calculateEuclideanDistance(input_vector, training_data[i]);
75  distances.emplace_back(distance, labels[i]);
76  }
77 
78  // Sort distances in ascending order
79  std::sort(
80  distances.begin(),
81  distances.end(),
82  [](const std::pair<double, int> &a, const std::pair<double, int> &b) {
83  return a.first < b.first;
84  });
85 
86  // Count votes for each label among the k nearest neighbors
87  std::unordered_map<int, int> label_counts;
88  for (int i = 0; i < k; ++i) {
89  label_counts[distances[i].second]++;
90  }
91 
92  // Find the label with the maximum votes
93  int max_votes = -1;
94  int predicted_label = -1;
95  for (const auto &entry : label_counts) {
96  if (entry.second > max_votes) {
97  max_votes = entry.second;
98  predicted_label = entry.first;
99  }
100  }
101 
102  return predicted_label;
103 }

105 double
106 gpmp::ml::KNN::calculateEuclideanDistance(const std::vector<double> &vec1,
107  const std::vector<double> &vec2) {
108  double distance = 0.0;
109  for (size_t i = 0; i < vec1.size(); ++i) {
110  double diff = vec1[i] - vec2[i];
111  distance += diff * diff;
112  }
113  return std::sqrt(distance);
114 }
